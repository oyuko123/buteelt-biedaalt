13      public static List<Card> loadCardsFromFile(String filename) throws IOException {
14          List<Card> cards = new ArrayList<>();
15          List<String> lines = Files.readAllLines(Paths.get(filename));
16          for (int i = 0; i < lines.size(); i += 2) {
17              if (i + 1 < lines.size()) {
18                  cards.add(new Card(lines.get(i), lines.get(i + 1)));
19              }
20          }
21          return cards;
22      }
