9   public class CommandLineOptionsTest {
10  
11      @Test
12      public void testDefaultOptions() {
13          CommandLineOptions options = new CommandLineOptions(new String[] {"file.txt"});
14          assertFalse(options.isHelp());
15          assertEquals("random", options.getOrder());
16          assertEquals(1, options.getRepetitions());
17          assertFalse(options.isInvertCards());
18          assertEquals("file.txt", options.getCardsFile());
19      }
20  
21      @Test
22      public void testHelpOption() {
23          CommandLineOptions options = new CommandLineOptions(new String[] {"--help"});
24          assertTrue(options.isHelp());
25          assertNull(options.getCardsFile());
26      }
27  
28      @Test
29      public void testOrderOption() {
30          CommandLineOptions options = new CommandLineOptions(
31                  new String[] {"--order", "worst-first", "file.txt"});
32          assertEquals("worst-first", options.getOrder());
33          
34          // Test invalid order
35          options = new CommandLineOptions(
36                  new String[] {"--order", "invalid-order", "file.txt"});
37          assertTrue(options.isHelp() || "random".equals(options.getOrder()));
38      }
39  
40      @Test
41      public void testRepetitionsOption() {
42          CommandLineOptions options = new CommandLineOptions(
43                  new String[] {"--repetitions", "3", "file.txt"});
44          assertEquals(3, options.getRepetitions());
45  
46          options = new CommandLineOptions(
47                  new String[] {"--repetitions", "-1", "file.txt"});
48          assertTrue(options.isHelp() || options.getRepetitions() == 1);
49          
50          options = new CommandLineOptions(
51                  new String[] {"--repetitions", "not-a-number", "file.txt"});
52          assertTrue(options.isHelp() || options.getRepetitions() == 1);
53      }
54  
55      @Test
56      public void testInvertCardsOption() {
57          CommandLineOptions options = new CommandLineOptions(
58                  new String[] {"--invertCards", "file.txt"});
59          assertTrue(options.isInvertCards());
60      }
61  
62      @Test
63      public void testMultipleOptions() {
64          CommandLineOptions options = new CommandLineOptions(
65                  new String[] {"--order", "recent-mistakes-first", "--repetitions", "5", 
66                                "--invertCards", "file.txt"});
67          assertEquals("recent-mistakes-first", options.getOrder());
68          assertEquals(5, options.getRepetitions());
69          assertTrue(options.isInvertCards());
70          assertEquals("file.txt", options.getCardsFile());
71      }
